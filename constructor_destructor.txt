#include "stdafx.h"
#include <iostream>
using namespace std;

class example
{
	int i,j;
	char z;
public:

	example()
	{
		cout<<"Default Consructor"<<"\n";
	}

	example(int n)
	{
		cout<<"This is a constructor 1"<<"\n";
	i=n;
	cout<<"The number is"<<i<<"\n";
	}

	~example()
	{
		cout<<"This is a destructor"<<"\n";
	}

	example(int n1, char a)
	{
		cout<<"This is a constructor 2"<<"\n";
	j=n1;
	z=a;
	cout<<"The number is"<<j<<"\n";
	cout<<"The char is :"<<a<<"\n";
	}
	void print()
	{
		cout<<"hi";
	}
};


int _tmain(int argc, _TCHAR* argv[])
{
	example e1;
	example e2(10); 
	example e3(4,'i');
	e1.print();
}
